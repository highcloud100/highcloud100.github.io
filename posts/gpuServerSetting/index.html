# Anaconda 설치 파일 다운로드

# 설치 파일 실행
bash Anaconda3-2020.11-Linux-x86_64.sh

# Anaconda 설치 후 재접속

# conda 환경 생성 예시
conda create -n <환경 이름> python=<파이썬 버전>

# conda 환경 실행
conda activate <환경 이름>
